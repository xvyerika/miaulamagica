        const cuentosData = [
            {t: "El Osito Valiente", c: "Había una vez un osito que aprendió a caminar solo por el bosque. Fue muy valiente y descubrió flores mágicas.", audio: "a1.mp3"},
            {t: "La Luna de Queso", c: "La luna brillaba tanto que un ratoncito quiso construir un cohete de cartón para subir y probarla.", audio: "a2.mp3"},
            {t: "La Gatita Aventurera", c: "Michi salió a explorar el jardín y encontró un tesoro escondido: una pelota que rebotaba hasta las nubes.", audio: "a3.mp3"},
            {t: "El Sol Alegre", c: "El sol salía cada mañana para despertar a todas las flores del campo con un abrazo de luz dorada.", audio: "a4.mp3"},
            {t: "El Árbol Sabio", c: "Había un árbol muy viejo que contaba historias a los pajaritos cada tarde cuando el viento soplaba.", audio: "a5.mp3"},
            {t: "El Pez de Colores", c: "Un pececito vivía en un río de cristal y soñaba con conocer el mar y ver las estrellas del fondo.", audio: "a6.mp3"}
        ];

        function abrirActividad(tipo) {
            document.querySelector('.contenedor-modulos').style.display = 'none';
            document.getElementById('pantalla-juego').style.display = 'block';
            let contenido = document.getElementById('contenido-actividad');
            contenido.innerHTML = ""; 

            if (tipo === 'cuentos') {
                mostrarListaCuentos();
            } else {
                contenido.innerHTML = "<h2>Contenido en construcción</h2>";
            }
        }

        function mostrarListaCuentos() {
            let contenido = document.getElementById('contenido-actividad');
            contenido.innerHTML = "<h2>Elige un Cuento</h2><div class='grid'>" + 
                cuentosData.map((h, i) => `<div class='carta' onclick='leerCuento(${i})'><h3>${h.t}</h3><p>Haz clic para leer</p></div>`).join("") + "</div>";
        }

        function leerCuento(index) {
            let contenido = document.getElementById('contenido-actividad');
            let cuento = cuentosData[index];
            contenido.innerHTML = `<button class='btn-volver' onclick='mostrarListaCuentos()'>⬅ Volver a la lista</button>
                                   <h2>${cuento.t}</h2>
                                   <audio controls id="miAudio" style="width:100%; margin-bottom:20px;">
                                       <source src="${cuento.audio}" type="audio/mpeg">
                                       Tu navegador no soporta el audio.
                                   </audio>
                                   <p style='font-size:1.5rem; line-height:1.6;'>${cuento.c}</p>`;
            
            // Forzar carga del navegador
            let audioPlayer = document.getElementById('miAudio');
            audioPlayer.load();

            // Activar reproducción con el primer clic del usuario
            document.body.onclick = function() {
                audioPlayer.play();
                document.body.onclick = null; 
            };
        }

        function volverAlMenu() {
            const audio = document.querySelector('audio');
            if (audio) { audio.pause(); audio.currentTime = 0; }
            document.querySelector('.contenedor-modulos').style.display = 'flex';
            document.getElementById('pantalla-juego').style.display = 'none';
        }
    </script>
